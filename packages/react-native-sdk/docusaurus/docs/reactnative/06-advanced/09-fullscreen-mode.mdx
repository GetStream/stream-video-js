[//]: # ()
[//]: # (In this guide, you'll learn how to display `ParticipantView` in a fullscreen mode. To make this work we'll be utilising [`element.requestFullscreen` browser API]&#40;https://developer.mozilla.org/en-US/docs/Web/API/Element/requestFullscreen&#41; and `ParticipantViewContext` to reach for `participantViewElement`. You might want to use this customization in combination with `SpeakerLayout` &#40;`ParticipantViewUISpotlight` property&#41; where spotlighted `ParticipantView` switches between dominant speakers automatically.)

[//]: # ()
[//]: # (### Define custom `ParticipantViewUI`)

[//]: # ()
[//]: # (The `ParticipantViewUI` template is incredibly versatile and can render anything you want it to. In this case, we'll be rendering a single button which we'll use to toggle between fullscreen and normal mode.)

[//]: # ()
[//]: # (```tsx)

[//]: # (import { useParticipantViewContext } from '@stream-io/video-react-sdk';)

[//]: # ()
[//]: # (const CustomParticipantViewUI = &#40;&#41; => {)

[//]: # (  const { participantViewElement } = useParticipantViewContext&#40;&#41;;)

[//]: # (  const [isFullsreenElement, setIsFullscreenElement] = useState&#40;false&#41;;)

[//]: # ()
[//]: # (  useEffect&#40;&#40;&#41; => {)

[//]: # (    // sync local state)

[//]: # (    const handleFullscreenChange = &#40;&#41; => {)

[//]: # (      setIsFullscreenElement&#40;)

[//]: # (        document.fullscreenElement === participantViewElement,)

[//]: # (      &#41;;)

[//]: # (    };)

[//]: # ()
[//]: # (    window.addEventListener&#40;'fullscreenchange', handleFullscreenChange&#41;;)

[//]: # ()
[//]: # (    return &#40;&#41; =>)

[//]: # (      window.removeEventListener&#40;'fullscreenchange', handleFullscreenChange&#41;;)

[//]: # (  }, [participantViewElement]&#41;;)

[//]: # ()
[//]: # (  const toggleFullscreen = &#40;&#41; => {)

[//]: # (    if &#40;isFullsreenElement&#41; {)

[//]: # (      return document.exitFullscreen&#40;&#41;;)

[//]: # (    })

[//]: # ()
[//]: # (    return participantViewElement?.requestFullscreen&#40;&#41;;)

[//]: # (  };)

[//]: # ()
[//]: # (  return &#40;)

[//]: # (    <>)

[//]: # (      <button)

[//]: # (        style={{ position: 'absolute', top: 10, right: 10 }})

[//]: # (        onClick={toggleFullscreen})

[//]: # (      >)

[//]: # (        {isFullsreenElement ? 'Leave' : 'Enter'} fullscreen)

[//]: # (      </button>)

[//]: # (      {/* your other custom UI elements */})

[//]: # (    </>)

[//]: # (  &#41;;)

[//]: # (};)

[//]: # (```)

[//]: # ()
[//]: # (## Final steps)
